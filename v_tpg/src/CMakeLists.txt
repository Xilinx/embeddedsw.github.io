cmake_minimum_required(VERSION 3.14.7)
project(v_tpg)

find_package(common)
collector_create (PROJECT_LIB_HEADERS "${CMAKE_CURRENT_SOURCE_DIR}")
collector_create (PROJECT_LIB_SOURCES "${CMAKE_CURRENT_SOURCE_DIR}")
include_directories(${CMAKE_BINARY_DIR}/include)
collect (PROJECT_LIB_SOURCES xv_tpg.c)
collect (PROJECT_LIB_SOURCES xv_tpg_g.c)
collect (PROJECT_LIB_HEADERS xv_tpg_hw.h)
collect (PROJECT_LIB_SOURCES xv_tpg_linux.c)
collect (PROJECT_LIB_HEADERS xv_tpg.h)
collect (PROJECT_LIB_SOURCES xv_tpg_sinit.c)
collector_list (_sources PROJECT_LIB_SOURCES)
collector_list (_headers PROJECT_LIB_HEADERS)
file(COPY ${_headers} DESTINATION ${CMAKE_BINARY_DIR}/include)
add_library(v_tpg STATIC ${_sources})
set_target_properties(v_tpg PROPERTIES LINKER_LANGUAGE C)
