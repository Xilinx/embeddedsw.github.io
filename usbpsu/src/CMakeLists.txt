cmake_minimum_required(VERSION 3.15)
project(usbpsu)

find_package(common)
collector_create (PROJECT_LIB_HEADERS "${CMAKE_CURRENT_SOURCE_DIR}")
collector_create (PROJECT_LIB_SOURCES "${CMAKE_CURRENT_SOURCE_DIR}")
include_directories(${CMAKE_BINARY_DIR}/include)
collect (PROJECT_LIB_SOURCES xusbpsu.c)
collect (PROJECT_LIB_HEADERS xusbpsu.h)
collect (PROJECT_LIB_SOURCES xusbpsu_controltransfers.c)
collect (PROJECT_LIB_SOURCES xusbpsu_endpoint.c)
collect (PROJECT_LIB_HEADERS xusbpsu_endpoint.h)
collect (PROJECT_LIB_SOURCES xusbpsu_g.c)
collect (PROJECT_LIB_SOURCES xusbpsu_hibernation.c)
collect (PROJECT_LIB_HEADERS xusbpsu_hw.h)
collect (PROJECT_LIB_SOURCES xusbpsu_intr.c)
collect (PROJECT_LIB_SOURCES xusbpsu_sinit.c)
collect (PROJECT_LIB_SOURCES xusbpsu_command.c)
collect (PROJECT_LIB_SOURCES xusbpsu_device.c)
collect (PROJECT_LIB_SOURCES xusbpsu_ep0handler.c)
collect (PROJECT_LIB_SOURCES xusbpsu_ephandler.c)
collect (PROJECT_LIB_SOURCES xusbpsu_event.c)
collect (PROJECT_LIB_HEADERS xusbpsu_local.h)
collector_list (_sources PROJECT_LIB_SOURCES)
collector_list (_headers PROJECT_LIB_HEADERS)
file(COPY ${_headers} DESTINATION ${CMAKE_BINARY_DIR}/include)
add_library(usbpsu STATIC ${_sources})
set_target_properties(usbpsu PROPERTIES LINKER_LANGUAGE C)
