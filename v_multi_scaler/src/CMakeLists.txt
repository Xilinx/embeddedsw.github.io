cmake_minimum_required(VERSION 3.14.7)
project(v_multi_scaler)

find_package(common)
collector_create (PROJECT_LIB_HEADERS "${CMAKE_CURRENT_SOURCE_DIR}")
collector_create (PROJECT_LIB_SOURCES "${CMAKE_CURRENT_SOURCE_DIR}")
include_directories(${CMAKE_BINARY_DIR}/include)
collect (PROJECT_LIB_HEADERS xv_multi_scaler_hw.h)
collect (PROJECT_LIB_HEADERS xv_multi_scaler_l2.h)
collect (PROJECT_LIB_SOURCES xv_multi_scaler_l2.c)
collect (PROJECT_LIB_SOURCES xv_multi_scaler_sinit.c)
collect (PROJECT_LIB_SOURCES xv_multi_scaler.c)
collect (PROJECT_LIB_SOURCES xv_multi_scaler_g.c)
collect (PROJECT_LIB_SOURCES xv_multi_scaler_coeff.c)
collect (PROJECT_LIB_SOURCES xv_multi_scaler_intr.c)
collect (PROJECT_LIB_HEADERS xv_multi_scaler.h)
collector_list (_sources PROJECT_LIB_SOURCES)
collector_list (_headers PROJECT_LIB_HEADERS)
file(COPY ${_headers} DESTINATION ${CMAKE_BINARY_DIR}/include)
add_library(v_multi_scaler STATIC ${_sources})
set_target_properties(v_multi_scaler PROPERTIES LINKER_LANGUAGE C)
