cmake_minimum_required(VERSION 3.14.7)

project(i2stx)

find_package(common)
collector_create (PROJECT_LIB_HEADERS "${CMAKE_CURRENT_SOURCE_DIR}")
collector_create (PROJECT_LIB_SOURCES "${CMAKE_CURRENT_SOURCE_DIR}")
include_directories(${CMAKE_BINARY_DIR}/include)
collect (PROJECT_LIB_SOURCES xi2stx_chsts.c)
collect (PROJECT_LIB_HEADERS xi2stx_chsts.h)
collect (PROJECT_LIB_SOURCES xi2stx_debug.c)
collect (PROJECT_LIB_HEADERS xi2stx_debug.h)
collect (PROJECT_LIB_SOURCES xi2stx_g.c)
collect (PROJECT_LIB_HEADERS xi2stx_hw.h)
collect (PROJECT_LIB_SOURCES xi2stx_intr.c)
collect (PROJECT_LIB_SOURCES xi2stx_selftest.c)
collect (PROJECT_LIB_SOURCES xi2stx_sinit.c)
collect (PROJECT_LIB_SOURCES xi2stx.c)
collect (PROJECT_LIB_HEADERS xi2stx.h)
collector_list (_sources PROJECT_LIB_SOURCES)
collector_list (_headers PROJECT_LIB_HEADERS)
file(COPY ${_headers} DESTINATION ${CMAKE_BINARY_DIR}/include)
add_library(i2stx STATIC ${_sources})
set_target_properties(i2stx PROPERTIES LINKER_LANGUAGE C)
