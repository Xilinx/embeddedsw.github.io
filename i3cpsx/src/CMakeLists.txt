cmake_minimum_required(VERSION 3.15)
project(i3cps)

find_package(common)
collector_create (PROJECT_LIB_HEADERS "${CMAKE_CURRENT_SOURCE_DIR}")
collector_create (PROJECT_LIB_SOURCES "${CMAKE_CURRENT_SOURCE_DIR}")
include_directories(${CMAKE_BINARY_DIR}/include)
collect (PROJECT_LIB_SOURCES xi3cpsx_g.c)
collect (PROJECT_LIB_SOURCES xi3cpsx.c)
collect (PROJECT_LIB_HEADERS xi3cpsx.h)
collect (PROJECT_LIB_SOURCES xi3cpsx_slave.c)
collect (PROJECT_LIB_SOURCES xi3cpsx_hw.c)
collect (PROJECT_LIB_SOURCES xi3cpsx_master.c)
collect (PROJECT_LIB_HEADERS xi3cpsx_hw.h)
collect (PROJECT_LIB_HEADERS xi3cpsx_pr.h)
collect (PROJECT_LIB_SOURCES xi3cpsx_options.c)
collect (PROJECT_LIB_SOURCES xi3cpsx_sinit.c)
collector_list (_sources PROJECT_LIB_SOURCES)
collector_list (_headers PROJECT_LIB_HEADERS)
file(COPY ${_headers} DESTINATION ${CMAKE_BINARY_DIR}/include)
add_library(i3cpsx STATIC ${_sources})
set_target_properties(i3cpsx PROPERTIES LINKER_LANGUAGE C)
