cmake_minimum_required(VERSION 3.15)
project(dfemix)

find_package(common)
add_definitions(-D__BAREMETAL__)
add_definitions(-DXLNX_PLATFORM)
collector_create (PROJECT_LIB_HEADERS "${CMAKE_CURRENT_SOURCE_DIR}")
collector_create (PROJECT_LIB_SOURCES "${CMAKE_CURRENT_SOURCE_DIR}")
include_directories(${CMAKE_BINARY_DIR}/include)
collect (PROJECT_LIB_SOURCES xdfemix.c)
collect (PROJECT_LIB_HEADERS xdfemix_hw.h)
collect (PROJECT_LIB_HEADERS xdfemix.h)
collect (PROJECT_LIB_SOURCES xdfemix_sinit.c)
collect (PROJECT_LIB_SOURCES xdfemix_intr.c)
collect (PROJECT_LIB_SOURCES xdfemix_g.c)
collector_list (_sources PROJECT_LIB_SOURCES)
collector_list (_headers PROJECT_LIB_HEADERS)
file(COPY ${_headers} DESTINATION ${CMAKE_BINARY_DIR}/include)
add_library(dfemix STATIC ${_sources})
set_target_properties(dfemix PROPERTIES LINKER_LANGUAGE C)
