cmake_minimum_required(VERSION 3.14.7)
project(dsi)

find_package(common)
collector_create (PROJECT_LIB_HEADERS "${CMAKE_CURRENT_SOURCE_DIR}")
collector_create (PROJECT_LIB_SOURCES "${CMAKE_CURRENT_SOURCE_DIR}")
include_directories(${CMAKE_BINARY_DIR}/include)
collect (PROJECT_LIB_SOURCES xdsi.c)
collect (PROJECT_LIB_HEADERS xdsi.h)
collect (PROJECT_LIB_SOURCES xdsi_g.c)
collect (PROJECT_LIB_HEADERS xdsi_hw.h)
collect (PROJECT_LIB_SOURCES xdsi_intr.c)
collect (PROJECT_LIB_SOURCES xdsi_selftest.c)
collect (PROJECT_LIB_SOURCES xdsi_sinit.c)
collector_list (_sources PROJECT_LIB_SOURCES)
collector_list (_headers PROJECT_LIB_HEADERS)
file(COPY ${_headers} DESTINATION ${CMAKE_BINARY_DIR}/include)
add_library(dsi STATIC ${_sources})
set_target_properties(dsi PROPERTIES LINKER_LANGUAGE C)
