cmake_minimum_required(VERSION 3.14.7)
project(v_warp_init)

find_package(common)
collector_create (PROJECT_LIB_HEADERS "${CMAKE_CURRENT_SOURCE_DIR}")
collector_create (PROJECT_LIB_SOURCES "${CMAKE_CURRENT_SOURCE_DIR}")
include_directories(${CMAKE_BINARY_DIR}/include)
collect (PROJECT_LIB_SOURCES xv_warp_init.c)
collect (PROJECT_LIB_HEADERS xv_warp_init.h)
collect (PROJECT_LIB_SOURCES xv_warp_init_g.c)
collect (PROJECT_LIB_HEADERS xv_warp_init_hw.h)
collect (PROJECT_LIB_SOURCES xv_warp_init_intr.c)
collect (PROJECT_LIB_SOURCES xv_warp_init_l2.c)
collect (PROJECT_LIB_HEADERS xv_warp_init_l2.h)
collect (PROJECT_LIB_SOURCES xv_warp_init_linux.c)
collect (PROJECT_LIB_SOURCES xv_warp_init_sinit.c)
collect (PROJECT_LIB_SOURCES xv_warp_init_utils.c)
collect (PROJECT_LIB_HEADERS xv_warp_init_utils.h)
collector_list (_sources PROJECT_LIB_SOURCES)
collector_list (_headers PROJECT_LIB_HEADERS)
file(COPY ${_headers} DESTINATION ${CMAKE_BINARY_DIR}/include)
add_library(v_warp_init STATIC ${_sources})
set_target_properties(v_warp_init PROPERTIES LINKER_LANGUAGE C)
