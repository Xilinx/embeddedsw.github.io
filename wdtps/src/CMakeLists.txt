cmake_minimum_required(VERSION 3.15)
project(wdtps)

find_package(common)
collector_create (PROJECT_LIB_HEADERS "${CMAKE_CURRENT_SOURCE_DIR}")
collector_create (PROJECT_LIB_SOURCES "${CMAKE_CURRENT_SOURCE_DIR}")
include_directories(${CMAKE_BINARY_DIR}/include)
collect (PROJECT_LIB_SOURCES xwdtps_selftest.c)
collect (PROJECT_LIB_SOURCES xwdtps_g.c)
collect (PROJECT_LIB_HEADERS xwdtps.h)
collect (PROJECT_LIB_HEADERS xwdtps_hw.h)
collect (PROJECT_LIB_SOURCES xwdtps_sinit.c)
collect (PROJECT_LIB_SOURCES xwdtps.c)
collector_list (_sources PROJECT_LIB_SOURCES)
collector_list (_headers PROJECT_LIB_HEADERS)
file(COPY ${_headers} DESTINATION ${CMAKE_BINARY_DIR}/include)
add_library(wdtps STATIC ${_sources})
set_target_properties(wdtps PROPERTIES LINKER_LANGUAGE C)
