cmake_minimum_required(VERSION 3.14.7)
project(dsitxss)

find_package(common)
collector_create (PROJECT_LIB_HEADERS "${CMAKE_CURRENT_SOURCE_DIR}")
collector_create (PROJECT_LIB_SOURCES "${CMAKE_CURRENT_SOURCE_DIR}")
include_directories(${CMAKE_BINARY_DIR}/include)
collect (PROJECT_LIB_SOURCES xdsitxss.c)
collect (PROJECT_LIB_HEADERS xdsitxss.h)
collect (PROJECT_LIB_SOURCES xdsitxss_g.c)
collect (PROJECT_LIB_HEADERS xdsitxss_hw.h)
collect (PROJECT_LIB_SOURCES xdsitxss_intr.c)
collect (PROJECT_LIB_SOURCES xdsitxss_selftest.c)
collect (PROJECT_LIB_SOURCES xdsitxss_sinit.c)
collector_list (_sources PROJECT_LIB_SOURCES)
collector_list (_headers PROJECT_LIB_HEADERS)
file(COPY ${_headers} DESTINATION ${CMAKE_BINARY_DIR}/include)
add_library(dsitxss STATIC ${_sources})
set_target_properties(dsitxss PROPERTIES LINKER_LANGUAGE C)
