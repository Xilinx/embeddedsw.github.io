cmake_minimum_required(VERSION 3.14.7)
project(mipicsiss)

find_package(common)
collector_create (PROJECT_LIB_HEADERS "${CMAKE_CURRENT_SOURCE_DIR}")
collector_create (PROJECT_LIB_SOURCES "${CMAKE_CURRENT_SOURCE_DIR}")
include_directories(${CMAKE_BINARY_DIR}/include)
collect (PROJECT_LIB_SOURCES xcsiss.c)
collect (PROJECT_LIB_HEADERS xcsiss.h)
collect (PROJECT_LIB_SOURCES xcsiss_g.c)
collect (PROJECT_LIB_HEADERS xcsiss_hw.h)
collect (PROJECT_LIB_SOURCES xcsiss_intr.c)
collect (PROJECT_LIB_SOURCES xcsiss_selftest.c)
collect (PROJECT_LIB_SOURCES xcsiss_sinit.c)
collector_list (_sources PROJECT_LIB_SOURCES)
collector_list (_headers PROJECT_LIB_HEADERS)
file(COPY ${_headers} DESTINATION ${CMAKE_BINARY_DIR}/include)
add_library(mipicsiss STATIC ${_sources})
set_target_properties(mipicsiss PROPERTIES LINKER_LANGUAGE C)
